select ROWNUM rn,a.*  from  (SELECT     DISTINCT
            c.ID,c.CATEGORY_NAME,count(r.QUESTION_CATEGORY_ID) AS QUESTIONCOUNT
            FROM QUESTION_CATEGORY c
            LEFT JOIN QUESTION_CATEGORY_REL r ON c.ID = r.QUESTION_CATEGORY_ID 
            GROUP BY c.ID,c.CATEGORY_NAME,r.QUESTION_CATEGORY_ID ) a
